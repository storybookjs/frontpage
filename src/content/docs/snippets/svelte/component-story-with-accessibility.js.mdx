```js
// Button.stories.js

import Button from './Button.svelte';

export default {
  component: Button,
  argTypes: {
    backgroundColor: { control: 'color' },
  },
};

// This is an accessible story
export const Accessible = {
  render: (args) => ({
    Component: Button,
    props: args,
  }),
  args: {
    primary: false,
    label: 'Button',
  },
};

// This is not
export const Inaccessible = {
  render: (args) => ({
    Component: Button,
    props: args,
  }),
  args: {
    ...Accessible.args,
    backgroundColor: 'red',
  },
};
```
