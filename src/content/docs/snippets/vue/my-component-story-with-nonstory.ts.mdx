  component: MyComponent,
  includeStories: ['SimpleStory', 'ComplexStory'],
  excludeStories: /.*Data$/, // ðŸ‘ˆ Storybook ignores anything that contains Data

export const SimpleStory: Story = {
  args: {
    data: simpleData,
  },
};

  args: {
    data: complexData,
  },
};
